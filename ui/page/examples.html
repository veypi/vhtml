条件渲染

🎉 这是条件渲染的内容！ 
<script setup>
  isVisible = true;
  toggleShow = () => {
    isVisible = !isVisible;
  };
</script>

<button @click="toggleShow">{{ isVisible ? '隐藏' : '显示' }}</button>
<div v-if="isVisible">🎉 这是条件渲染的内容！</div>

列表渲染

  • {{ todo.text }} 
<script setup>
  newTodo = '';
  todos = [
    { text: '学习 vhtml', done: true },
    { text: '创建项目', done: false }
  ];

  addTodo = () => {
    if (newTodo.trim()) {
      todos.push({ text: newTodo, done: false });
      newTodo = '';
    }
  };

  toggleTodo = (index) => {
    todos[index].done = !todos[index].done;
  };

  removeTodo = (index) => {
    todos.splice(index, 1);
  };
</script>

<li v-for="(todo, index) in todos"
    :class="{completed: todo.done}">
  <input type="checkbox"
         :checked="todo.done"
         @change="toggleTodo(index)">
  <span>{{ todo.text }}</span>
</li>

Tab 切换

👤 这是个人资料内容

<script setup>
  activeTab = 'home';
  switchTab = (tab) => {
    activeTab = tab;
  };
</script>

<button :class="{active: activeTab === 'home'}"
        @click="switchTab('home')">首页</button>
<div v-if="activeTab === 'home'">🏠 首页内容</div>
